// Load surprise archive links (links 9-33) async function loadSurpriseLinks() { try { const response = await fetch('/surprise-me-links.txt'); if (!response.ok) { console.error('Failed to load surprise-me-links.txt:', response.status); document.getElementById('archive-error').style.display = 'block'; return; } const text = await response.text(); const lines = text.split('\n').filter(line => line.trim()); const links = lines.map(line => { const parts = line.split('|'); if (parts.length >= 2) { return { title: parts[0].trim(), url: parts[1].trim() }; } return null; }).filter(link => link !== null); // Take links 9-33 (skip first 8, show next 25) const archiveLinks = links.length > 8 ? links.slice(8, 33) : []; const grid = document.getElementById('surprise-links'); if (grid && archiveLinks.length > 0) { grid.innerHTML = archiveLinks.map(link => `
  • ${link.title}
    • `).join(''); } else { console.log('No archive links available (need more than 8 links in surprise-me-links.txt)'); document.getElementById('archive-error').textContent = 'No archived articles available to display.'; document.getElementById('archive-error').style.display = 'block'; } console.log(`Showing ${archiveLinks.length} surprise archive links (9-33)`); } catch (error) { console.error('Error loading surprise links:', error); document.getElementById('archive-error').style.display = 'block'; } }
